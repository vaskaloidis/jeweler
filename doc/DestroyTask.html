def call
  if task.update(deleted: true)
    Note.create_event(task.sprint.project, 'task_deleted', 'Deleted: ' + task.description)
  else
    task.errors.full_messages.map { |e| fatals << 'Error Destroying Task: ' + e }
  end
  task
end
